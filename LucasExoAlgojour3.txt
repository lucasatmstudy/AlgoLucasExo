-Calcule le montant de la TVA
-Retourne ce montant

2 . Ensuite écrire un algorithme principale qui !

function calculerTVA(prixHT, tauxTVA)
	prixTVA <- prixHT * tauxTVA / 100
	Return prixTVA
FIN

Exercice 3 : Composer plusieurs fonctions
1 . Créer deux fonction :
	-'calculerMoyenne(note1, note2, note3)' qui retourne la moyenne de 3 notes
	-'attribuerMention(moyenne)' qui retourne une mention selon la moyenne :
		+moyenne >= 16 : "très bien
		+moyenne >= 14 : "Bien"
		+moyenne >= 12 : "Assez bien"
		+moyenne >= 10 : "Passable"
		+moyenne < 10 : "Insufisant"

2 . Ensuite, écrire un algorithme principale qui :
	-Demande 3 notes
	-Calcule la moyenne avec la première fonctoin
	-Obtenir la mention avec la deuxième fonction
	-Affiche moyenne et mention

function calculerMoyenne(note1, note2, note3)
	moyenne <- (note1 + note2 + note3)/3
	RETURN moyenne
END

function attribuerMention(moyenne)
	IF moyenne >= 16 & moyenne <= 20
		RETOURNER "Très bien"
	ELSE IF moyenne >= 14 & moyenne < 16
		RETOURNER "Bien"
        ELSE IF moyenne >= 12 & moyenne < 14
		RETOURNER "Assez bien"
        ELSE IF moyenne >= 12 & moyenne < 10
		RETOURNER "Passable
	ELSE IF moyenne < 10 & moyenne >= 0
		RETOURNER "Insuffisant"
	ELSE 
		RETOURNER "Données incorrect"
	END IF
END

DEMANDER noteA, noteB, noteC
noteA <- saisie utilisateur
noteB <- saisie utilisateur
noteC <- saisie utilisateur
LaMoyenne <- calculerMoyenne(noteA, noteB, noteC)
LaMention <- attribuerMention(LaMoyenne)
AFFICHER LaMoyenne + LaMention
